正则表达式创建的两种方式

let reg = /\d+/;

let reg2 = new RegExp(“\d+”)

正则表达式由两部分组成

  • 元字符

  • 修饰符

    常用的元字符
    //=>1.量词的元字符:设置出现的次数
    * 零到多次
    + 一到多次
    ? 零次或者一次
    {n} 出现n次
    {n,} 出现n到多次
    {n,m} 出现m到m次

    //=2.特殊元字符 单个或者组合在一起代表特殊的含义
    \ 转义字符(普通->特殊->普通)
    . \n(换行符) 以外的任意字符
    ^ 以哪一个元字符作为开始
    $ 以哪一个元字符作为结束
    \n 换行符
    \d 09 之间的一个数字
    \D 非 0    9 之间的一个数字
    \w 数字 字母 下划线中的任意一个字符
    \s 一个空白字符
    \t 一个制表符(一个TAB键：四个空格)
    x|y x或者y中的一个字符
    [xyz] x或者y或者z中的一个字符
    [^xy] 除 x/y 以外的任意字符
    [a-z] 指 a到z的任意字符 [0-9a-zA-Z_] === \W
    [^a-z] 上一个的取反非
    () 正则中的分组符号
    (?:) 只匹配不获取
    (?=) 正向预查
    (?!) 负向预查

元字符详细解析

^ $

let reg = /^\d/ 只能以数字开头
括号中出现的字符一般都代表本身的含义
let reg = /^[@+]+$/ +代表+ 不是一个或多个的意思
let reg = /^[\d]+$/ \d在中括号中还是代表0-9(特殊)
中括号中不存在多位数
reg = /^[10-29]$/ 1或者0-2或者9

常用的正则表达式

  • 验证是否为有效数字

    规则分析
    1 可能出现正负号 也可能不出现 [+-]?
    2 一位0-9都可以多位首位不能为0 (\d|([1-9]\d+))
    3 小数部分可能有可能没有 一旦有后面必须有小数点+数字(.\d+)
    let reg = /^(+|-)?(\d|([1-9]\d+))(.\d+)?$/ 或者/^[+-]?$/

  • 验证密码

    let reg = /^\w{6,12}$/;

  • 验证真实姓名

    1 汉字 /^[\u4E00-\u9FA5]$/
    2 名字长度2到10位
    3 可能有译名 汉字
    let reg = /^[\u4E00-\u9FA5]{2,10}(·[\u4E00-\u9FA5]){0,2}$/

  • 验证邮箱

    let reg = /^\w+((-\w)|(.\w+))@[A-Za-z0-9]+((.|-)[A-Za-z0-9]+).[A-Za-z0-9]+$/;

  • 身份证号码

    十八位： ^[1-9]\d{5}(18|19|([23]\d))\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$
    十五位： ^[1-9]\d{5}\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{2}$
    总：
    ( ^[1-9]\d{5}(18|19|([23]\d))\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$)|(^[1-9]\d{5}\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{2}$)

正则表达式之两种创建方式的区别

构造函数因为传递的是字符串 ,\需要写两个才表示斜杠
let reg = /\d+/g
reg = new RegExp(“\d+”,”g”);

正则表达式中部分内容是变量存储的值
两个斜杠中间包起来的都是元字符(如果正则中想要包含某个变量的值,则不能使用字面量的创建方式)
let type =”jean”;
reg = /^@”+type+”@$/;
console.log(reg.test(“@jean@”));//false
console.log(reg.test(‘@”””typeeeee”@’));//true

reg = new RegExp(“^@”+type+”@$”);
console.log(reg.test(“@jean@”));//true

基于exec 实现的正则捕获
1.捕获的到结果是null或者是一个数组
第一项 是捕获到的内容
其余项:对应小分组本次单独捕获的内容
index:当前捕获内容在字符串中的索引
input:原始字符串
2.每执行一次exec.只能捕获到第一个符合正则规则的,但是默认情况下,我们执行一百遍,获取的永远是第一个匹配到的,其余捕获不到
=> 正则捕获的懒惰性:只捕获第一个
console.log(reg.exec(str));//[“2020”, index: 12, input: “zhangyingjun2020yanfdao2030”, groups: undefined]
reg.lastIndex : 当前正则下一次匹配的骑士索引位置
console.log(reg.lastIndex);// 0 下面匹配捕获的是从STR索引的位置开始找
console.log(reg..exec(str));
console.log(reg.lastIndex);// 0 第一次匹配捕获完成,lastIdex 没有改变,所以下一次exec依然是从字符串最开始找,找到的永远是第一个的
解决办法 全局捕获 g

let reg = /\d+/g;
console.log(reg.exec(str));//[“2020”,..]
console.log(reg.lastIndex);// 13 设置全局匹配修饰符g后,第一次匹配完,lastIndex会自己修改
console.log(reg.exec(str));//[“2030”,..]
console.log(reg.lastIndex);// 24
console.log(reg.exec(str));//null 当全局捕获后,再次捕获的结果是null,但是lastInedex又回归了初始值零,再次捕获又第一个开始了…
console.log(reg.lastIndex);//0
console.log(reg.exec(str));//[“2020”,..]

实现正则捕获的前提是:当前正则要和字符串匹配,如果不匹配捕获结果是null
console.log(reg.test(str));//false
console.log(reg.exec(str));//null

需求:编写一个方法execAll执行一次可以把所有的结果捕获到(前提正则一定要设置全局修饰符g)

~ function(){
function execAll(str=””){
//str:要匹配的字符串
//this:RegExp的实例（当前操作的正则）
//进来的第一件事,验证当前正则是否设置了G 不设置则不能再进行循环捕获了.否则会导致死循环
//ARY存储最后所捕获的信息 RES存储每一次捕获到的内容(数组)
if (!this.global) return this.exec(str);
let ary = [];
res = this.exec(str);
while (res){
//把每一次捕获的内容RES[0]存到数组中
ary.push(res[0]);
//只要捕获的内容不为NULL则继续捕获下去
res = this.exec(str);
}
return ary.length === 0 ? null : ary;
}
RegExp.prototype.execAll = execAll;
}();
字符串中的MATCH方法，可以在执行一次的情况下捕获到所有匹配的数据，前提正则也得设置G才行

正则的分组捕获

身份证号
let str = “1223231312312313123”;
let reg = /^(\d{6})(\d{4})(\d{2})(\d{2})(\d{2})(\d)(\d|X)$/
第一项 大正则匹配的结果
其余项:每一个小分组单独匹配的结果
如果设置了分组(改变优先级) 但是捕获的时候不需要单独捕获,可以基于?:来处理

match只能把大正则匹配的内容获取到,小分组匹配的信息无法获取

分组引用

let str = “book”;
let reg = /^a-zA-Z\1[a-zA-Z]$/
分组引用就是通过数字让其代表和对应分组出现一模一样的内容

正则捕获的贪婪性

let str = “asd2020@2030”;
正则捕获的贪婪性 默认情况下 正则捕获的时候是按照当前正则所匹配的最长结果来获取的
let reg = /\d+/g;
console.log(str.match(reg))// [“2020”,”2030”];
//取消懒惰性 在量词元字符后面设置 ?
let reg = /\d+?/g;
console.log(str.match(reg))// [“2”,”0”,”2”,”0”,”2”,”0”,”3”,”0”];

  • 问号左边是非量词元字符 本省代表量词元字符 出现零到一次
  • 问号左边是量词元字符 取消捕获的贪婪性
  • (?:)只匹配不捕获
  • (>=)正向预查
  • (?!)负向预查
